üöÄ Quick Start

# Install WebPilot
pip install claude-webpilot

# Basic usage
from webpilot import WebPilot

with WebPilot() as pilot:
    pilot.start("https://example.com")
    pilot.click(text="Login")
    pilot.type_text("username", selector="#email")
    pilot.screenshot("result.png")
