Installation (Standard Linux/macOS/Windows)

cd claude-webpilot

# Install with Poetry (recommended)

poetry run playwright install firefox

# Option 2: Native with FHS environment (See NIXOS_NATIVE_SOLUTION.md)

Python API

from src.webpilot.core import PlaywrightAutomation

# Modern Playwright API
with PlaywrightAutomation() as browser:
    browser.navigate("github.com")
    browser.click("Sign in")  # Auto-waits!
    browser.type_text("#username", "myuser")
    browser.screenshot("github_login")

# Backward compatible API (also uses Playwright!)
from src.webpilot.core import RealBrowserAutomation
browser = RealBrowserAutomation()
browser.start()
browser.navigate("example.com")
browser.close()

Network Interception Example

from src.webpilot.core import PlaywrightAutomation

with PlaywrightAutomation() as browser:
    # Enable network logging
    browser.enable_network_logging()

    # Navigate and interact
    browser.navigate("https://api.example.com")
    browser.click("Load Data")

    # Get all network requests
    logs = browser.get_network_logs()
    for log in logs:
        if log['type'] == 'request':
            print(f"{log['method']} {log['url']}")

    # Block resources for faster tests
    browser.block_resources(['image', 'stylesheet'])
    browser.navigate("example.com")  # Much faster!

üåê Multi-Browser Testing

Test your site on Firefox, Chromium, AND WebKit with the same code:

from src.webpilot.core.multi_browser import MultiBrowserTester

# Test on all browsers automatically
tester = MultiBrowserTester()
results = tester.test_url_on_all_browsers("https://example.com")

for browser, result in results.items():
    print(f"{browser}: {result['load_time']:.2f}s - {result['status']}")
    print(f"  Title: {result['title']}")

# Test specific viewports (responsive design)
viewports = [
    {'width': 375, 'height': 667},   # Mobile
    {'width': 768, 'height': 1024},  # Tablet
    {'width': 1920, 'height': 1080}  # Desktop
]

responsive_results = tester.test_responsive_design(
    "https://example.com",
    viewports
)
